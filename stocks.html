# Some useful operators, imports

import pandas
import mydl
import myplots
from sklearn.neighbors import KNeighborsRegressor
from sklearn.metrics import mean_squared_error, r2_score
import seaborn as sns
sns.set(rc={"figure.figsize": (20, 20)})

def fut(series, periods):
    return series.shift(-periods)

def pct(series, start_i, periods):
    return 100*(fut(series, start_i + periods) - fut(series, start_i))/(fut(series, start_i)*periods)

def frac(series, start_i, periods):
    return fut(series, start_i+periods)/fut(series, start_i)

def avg(series, start_i, periods):
    integral = sum(fut(series, start_i+x) for x in range(periods))
    return integral/abs(periods)

def norm(series):
    return (series-series.mean())/series.std()

# Learning

data = pandas.DataFrame({
    'S&P500 next 12m': pct(src['S&P500'], 0, 12),
    'S&P500 next 6m': pct(src['S&P500'], 0, 6),
    'S&P500 next 3m': pct(src['S&P500'], 0, 3),
    'p/e growth': frac(src['S&P500'], -48, 48)/(src['Earnings (USA)']/avg(src['Earnings (USA)'],-48,48)),
    'p/e': pct(src['S&P500'], -90, 90)/(src['Earnings (USA)']/avg(src['Earnings (USA)'],-90,90)),
    'moving avg': src['S&P500']/avg(src['S&P500'], -7, 7),
    'Overconfidence': norm(src['Business confidence (USA)']) - norm(frac(src['Earnings (USA)'],-36,36)),
})

# First order prediction
train = data.dropna()
feature_cols = ['moving avg','Overconfidence', 'p/e growth', 'p/e']
predict_cols = ['S&P500 next 6m','S&P500 next 3m', 'S&P500 next 12m']
models = dict()
N_NEIGHBORS = 5

for predict_col in predict_cols:
    X, y = (train.loc[:, feature_cols], train[predict_col])
    models[predict_col] = KNeighborsRegressor(n_neighbors=N_NEIGHBORS)
    models[predict_col].fit(X, y)

print("Done")
